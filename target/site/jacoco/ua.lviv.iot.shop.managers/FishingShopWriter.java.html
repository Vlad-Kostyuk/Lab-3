public class FishingShopWriter {

    private List<String> fishingShopWrite = new ArrayList<>();

    public void wtiteToFile(List<String> fishing) throws IOException{

        Bait bait = new Bait();

        FishingRods fishingRods = new FishingRods();

        fishingShopWrite = fishing;
        fishingShopWrite.add(fishingShop.getHeaders());
        fishingShopWrite.add(fishingShop.toCVS());

        String csvFile = "D/work/country.csv";
 
        File file = new File(csvFile);
        
        try {
            if (!file.exists()) {
                file.createNewFile();
            }
            FileWriter writer = new FileWriter(file); 
            BufferedWriter bw = new BufferedWriter(writer);
            bw.write(fishingShopWrite.toString());
            
            bw.flush();
            bw.close();

            e.printStackTrace();
        }
